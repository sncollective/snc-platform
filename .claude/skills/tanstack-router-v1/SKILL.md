# TanStack Router Reference

> **Version:** 1.x
> **Docs:** https://tanstack.com/router/latest

## Imports

```typescript
// Core
import { createFileRoute, createRootRouteWithContext, Link, Outlet, Navigate } from '@tanstack/react-router'

// Hooks
import { useNavigate, useParams, useSearch, useLocation } from '@tanstack/react-router'

// Utilities
import { getRouteApi, redirect, notFound, linkOptions } from '@tanstack/react-router'

// Vite Plugin
import { tanstackRouter } from '@tanstack/router-plugin/vite'

// Generated (auto-generated by plugin)

```

## API Quick Reference

### Route Definition

```typescript
// File-based route: routes/posts.$id.tsx
export const Route = createFileRoute('/posts/$id')({
  validateSearch: (search) => searchSchema.parse(search),
  parseParams: (params) => ({ id: params.id }),

  beforeLoad: ({ context, location }) => {
    if (!context.auth.isAuthenticated) {
      throw redirect({ to: '/login', search: { redirect: location.href } })
    }
  },

  loader: async ({ params, context }) => {
    return context.queryClient.ensureQueryData(postQuery(params.id))
  },

  component: PostPage,
  errorComponent: ({ error }) => <div>Error: {error.message}</div>,
  notFoundComponent: () => <div>Post not found</div>,
  pendingComponent: () => <div>Loading...</div>,
})
```

### Router Setup

```typescript
interface RouterContext {
  queryClient: QueryClient
}

export function createAppRouter(queryClient: QueryClient) {
  return createRouter({
    routeTree,
    context: { queryClient },
    defaultPreload: 'intent',
    defaultPreloadDelay: 100,
    scrollRestoration: true,
  })
}

// Type registration (required for full type safety)
declare module '@tanstack/react-router' {
  interface Register {
    router: ReturnType<typeof createAppRouter>
  }
}
```

### Search Parameters

```typescript
const searchSchema = z.object({
  page: z.coerce.number().int().min(1).default(1),
  query: z.string().optional(),
  filters: z.array(z.string()).optional(),
})

export const Route = createFileRoute('/search')({
  validateSearch: (search) => searchSchema.parse(search),
})

function SearchPage() {
  const { page, query, filters } = Route.useSearch()
}
```

### Data Loading

```typescript
// Preload in loader, use in component
export const Route = createFileRoute('/posts')({
  loader: ({ context }) =>
    context.queryClient.ensureQueryData(postsQuery),
})

function PostsPage() {
  const { data } = useSuspenseQuery(postsQuery)
  return <PostList posts={data} />
}

// Parallel loading
loader: async ({ context }) => {
  const [posts, user] = await Promise.all([
    context.queryClient.ensureQueryData(postsQuery),
    context.queryClient.ensureQueryData(userQuery),
  ])
  return { posts, user }
}

// Access loader data
const loaderData = Route.useLoaderData()
```

### Route Guards

```typescript
// Layout route: routes/_authenticated.tsx
export const Route = createFileRoute('/_authenticated')({
  beforeLoad: ({ context, location }) => {
    if (!context.auth.isAuthenticated) {
      throw redirect({
        to: '/login',
        search: { redirect: location.href },
      })
    }
  },
})
// All routes under _authenticated/ are protected
```

### Error Handling

```typescript
import { notFound } from '@tanstack/react-router'

loader: async ({ params }) => {
  const post = await fetchPost(params.id)
  if (!post) throw notFound()
  return post
}

// Router-level defaults
const router = createRouter({
  routeTree,
  defaultErrorComponent: ({ error }) => <div>Error: {error.message}</div>,
  defaultNotFoundComponent: () => <div>404</div>,
  notFoundMode: 'fuzzy',
})
```

### Code Splitting

```typescript
// Automatic (via Vite plugin)
tanstackRouter({ autoCodeSplitting: true })

// Manual: Split into critical (.tsx) and lazy (.lazy.tsx)
// routes/posts.$id.tsx (critical - loader, validators)
export const Route = createFileRoute('/posts/$id')({
  loader: ({ params }) => fetchPost(params.id),
})

// routes/posts.$id.lazy.tsx (lazy - component)
export const Route = createLazyFileRoute('/posts/$id')({
  component: PostPage,
})

// Access split routes via getRouteApi
import { getRouteApi } from '@tanstack/react-router'
const route = getRouteApi('/posts/$id')
function Component() {
  const data = route.useLoaderData()
  const params = route.useParams()
}
```

### File-Based Routing Conventions

```
routes/
├── __root.tsx           → Root layout (required)
├── index.tsx            → / (exact match)
├── about.tsx            → /about
├── posts.$id.tsx        → /posts/:id
├── posts.index.tsx      → /posts (exact)
├── posts.new.tsx        → /posts/new
├── _authenticated.tsx   → Pathless layout (no URL segment)
├── _authenticated/
│   ├── dashboard.tsx    → /dashboard (inherits layout)
│   └── settings.tsx     → /settings (inherits layout)
└── $.tsx                → Catch-all route
```

**Special prefixes:**
- `__root.tsx` - Root layout
- `index.tsx` - Exact route match
- `$param` - Path parameter
- `_layout` - Pathless layout route
- `$` - Catch-all
- `-file.tsx` - Ignored by router

## Gotchas & Version Caveats

### Always use `from` for type narrowing

```typescript
// Slow - TypeScript checks ALL routes
const navigate = useNavigate()

// Fast - TypeScript only checks relevant routes
const navigate = useNavigate({ from: Route.fullPath })
```

### Don't update URL on every keystroke

```typescript
// Bad - causes performance issues
<input onChange={(e) => navigate({ search: { q: e.target.value } })} />

// Good - local state + debounced URL update
const [draft, setDraft] = useState('')
const debouncedUpdate = useDebouncedCallback(
  (value) => navigate({ search: { q: value } }),
  300
)
```

### Router plugin MUST come before React plugin

```typescript
// vite.config.ts
export default defineConfig({
  plugins: [
    tanstackRouter(),  // First
    react(),           // Second
  ],
})
```

### Type registration required

```typescript
declare module '@tanstack/react-router' {
  interface Register {
    router: typeof router
  }
}
```

### notFound() in beforeLoad

Throwing `notFound()` in `beforeLoad` always triggers root notFoundComponent, not the route-level one.

### Search param serialization

TanStack Router uses JSON serialization for search params by default. Arrays, objects, booleans work automatically.

## Anti-Patterns

### String concatenation in paths
```typescript
// Bad
navigate({ to: `/posts/${id}` })

// Good
navigate({ to: '/posts/$id', params: { id } })
```

### Manual URL parsing
```typescript
// Bad
const params = new URLSearchParams(window.location.search)
const page = parseInt(params.get('page') || '1')

// Good - use validateSearch with Zod
```

### Not using Route.useX hooks
```typescript
// Bad - loses type safety
const params = useParams({ strict: false })

// Good
const params = Route.useParams()
```
